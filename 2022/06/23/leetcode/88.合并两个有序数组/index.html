class Solution {
public:
    // nums1 = [1,2,3,0,0,0], m = 3, nums2 = [2,5,6], n = 3
    void merge(vector<int>& nums1, int m, vector<int>& nums2, int n) {
        int i = m - 1, j = n - 1, tar = m + n - 1;
        while (j >= 0) {
            nums1[tar--] = (i >= 0 && nums1[i] > nums2[j] ? nums1[i--] : nums2[j--]);
        }

    }
};
