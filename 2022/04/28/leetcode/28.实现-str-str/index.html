func strStr(haystack string, needle string) int {
	if len(needle) == 0 {
		return 0
	}
	i, j := 0, 0
	for j < len(haystack) {

		c := needle[i]
		ch := haystack[j]
		if c == ch {
			i++
			j++
		} else {
			j = j - i + 1
			i = 0
		}
		if i == len(needle) {
			return j - i
		}

	}
	return -1
}
